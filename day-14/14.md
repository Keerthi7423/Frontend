Email
Password

↓

Without authentication:

Anyone can access protected pages ❌

6) Context API For Authentication

File:

AuthContext.jsx

Code:

import { createContext, useState } from "react";

export const AuthContext = createContext();

function AuthProvider({ children }) {

  const [isLoggedIn, setIsLoggedIn] = useState(false);

  function login() {
    setIsLoggedIn(true);
  }

  function logout() {
    setIsLoggedIn(false);
  }

  return (
    <AuthContext.Provider
      value={{
        isLoggedIn,
        login,
        logout
      }}
    >
      {children}
    </AuthContext.Provider>
  );
}

export default AuthProvider;

--------------------------------------------------

Code Explanation

createContext()
↓
Creates global auth storage

useState(false)
↓
Stores login status

login()
↓
Sets status to true

logout()
↓
Sets status to false

Provider
↓
Shares auth data globally

7) Protected Route

File:

ProtectedRoute.jsx

Code:

import { useContext } from "react";
import { Navigate } from "react-router-dom";
import { AuthContext } from "../context/AuthContext";

function ProtectedRoute({ children }) {

  const { isLoggedIn } =
    useContext(AuthContext);

  if (!isLoggedIn) {
    return <Navigate to="/" />;
  }

  return children;
}

export default ProtectedRoute;

--------------------------------------------------

Code Explanation

useContext(AuthContext)
↓
Reads login state

if (!isLoggedIn)
↓
User not authenticated

Navigate
↓
Redirect to login page

Else
↓
Render dashboard

8) Login Page

File:

Login.jsx

Code:

import { useContext } from "react";
import { useNavigate } from "react-router-dom";
import { AuthContext } from "../context/AuthContext";

function Login() {

  const { login } =
    useContext(AuthContext);

  const navigate = useNavigate();

  function handleLogin() {
    login();
    navigate("/dashboard");
  }

  return (
    <div>
      <h1>Login Page</h1>

      <button onClick={handleLogin}>
        Login
      </button>
    </div>
  );
}

export default Login;

--------------------------------------------------

Code Explanation

Button Click
↓
handleLogin()
↓
login()
↓
isLoggedIn becomes true
↓
navigate()
↓
Dashboard opens

9) Dashboard Page

File:

Dashboard.jsx

Code:

import { useContext } from "react";
import { useNavigate } from "react-router-dom";
import { AuthContext } from "../context/AuthContext";

function Dashboard() {

  const { logout } =
    useContext(AuthContext);

  const navigate = useNavigate();

  function handleLogout() {
    logout();
    navigate("/");
  }

  return (
    <div>
      <h1>Dashboard</h1>

      <button onClick={handleLogout}>
        Logout
      </button>
    </div>
  );
}

export default Dashboard;

--------------------------------------------------

Code Explanation

Logout button clicked
↓
logout()
↓
isLoggedIn becomes false
↓
navigate("/")
↓
Back to Login page

10) App.jsx

Code:

import { Routes, Route }
from "react-router-dom";

import AuthProvider
from "./context/AuthContext";

import Login from "./pages/Login";
import Dashboard from "./pages/Dashboard";

import ProtectedRoute
from "./components/ProtectedRoute";

function App() {

  return (
    <AuthProvider>

      <Routes>

        <Route
          path="/"
          element={<Login />}
        />

        <Route
          path="/dashboard"
          element={
            <ProtectedRoute>
              <Dashboard />
            </ProtectedRoute>
          }
        />

      </Routes>

    </AuthProvider>
  );
}

export default App;

--------------------------------------------------

Code Explanation

Routes
↓
Defines application pages

"/"
↓
Login page

"/dashboard"
↓
Protected dashboard page

ProtectedRoute
↓
Checks authentication
